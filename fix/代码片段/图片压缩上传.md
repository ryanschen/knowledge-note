```html
<input
  type="file"
  accept="image/*"
  class="uploader-input"
  @change="imgChange($event, 'right')"
  :multiple="false"
  @click="uploaderInputClick"
>
```
```js
export default {
  methods: {
    uploaderInputClick (e) {
      /**
       * fix: 第二次选择为同一图片，不触发change事件
       */
      e.target.value = null
    },

    uploadImgGetInfo (url, data) {
      return new Promise((resolve, reject) => {
        this.$toast.loading('请求中...', -1)
        $.post(
          url,
          data,
          { headers: { 'Content-Type': 'form-data' } }
        )
          .then(res => resolve(res))
          .catch(err => reject(err))
      })
    },

    getFormData (type, file) {
      let formData = new FormData()

      formData.append('orderNo', this.orderNo)
      formData.append('imageType', type)
      formData.append('files', file)

      return formData
    },

    dataURLtoFile (dataurl, filename) {
      let arr = dataurl.split(',')
      let mime = arr[0].match(/:(.*?);/)[1]
      let bstr = atob(arr[1])
      let n = bstr.length
      let u8arr = new Uint8Array(n)
      while (n--) {
        u8arr[n] = bstr.charCodeAt(n)
      }
      return new File([u8arr], filename, { type: mime })
    },

    fileToImage (file) {
      const reader = new FileReader()
      reader.readAsDataURL(file)
      return new Promise((resolve, reject) => {
        reader.onload = function (event) {
          const img = new Image()
          if (event.target.result) {
            img.onload = function () {
              resolve(img)
            }
            img.src = event.target.result // base64
          } else {
            reject(new Error('fileToImage error'))
          }
        }
      })
    },

    imgChange (e, type) {
      const { files } = e.target
      if (!files.length) return

      const f = files[0]
      // 小于300KB不进行压缩
      if (f.size < 300 * 1024) {
        this.uploadIDCard(type, f, e)
      } else {
        this.fileToImage(files[0])
          .then(img => {
            const canvas = document.createElement('canvas')
            const context = canvas.getContext('2d')
            canvas.width = 1200
            // 高度进行等比转换
            const h = parseInt(1200 * (img.height / img.width))
            canvas.height = h
            context.drawImage(img, 0, 0, 1200, h)
            const file = this.dataURLtoFile(
              canvas.toDataURL('image/jpeg'), // imgBase64
              `${new Date().getTime()}.jpeg`
            )
            this.uploadIDCard(type, file, e)
          })
          .catch(err => { console.log(err) })
      }
    },

    uploadIDCard (type, file, e) {
      this.uploadImgGetInfo(
        '/uploadIDCard',
        this.getFormData(
          type === 'left'
            ? 'frontIDCard'
            : 'reverseIDCard',
          file
        )
      )
        .then(res => {
          this.idCardInfo = {
            ...this.idCardInfo,
            ...JSON.parse(
              JSON.stringify(res.result, (k, v) => {
                if (!v) return undefined
                return v
              })
            )
          }
          const url = window.URL ||
            window.webkitURL ||
            window.mozURL

          this[`${type}IdCardImgSrc`] = url
            ? url.createObjectURL(file)
            : e.target.result
        })
        .catch(err => console.log(err))
    }
  }
}
```

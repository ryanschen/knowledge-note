Q：我现在有一个背包，容量为m，然后有n个货物，重量分别为w1,w2,w3...wn，每个货物的价值是v1,v2,v3...vn，w和v没有任何关系，请求背包能装下的最大价值？
A：
```js
var maxLoad = 15;
var loads = [5, 6, 2, 1, 9, 3, 5]
var prices = [4, 2, 5, 3, 2, 6, 7]

function maxValue(arr, m, load, price) {
  console.log(arr, m, load, price)
  for (var i = 0; i < arr.length - 1; i++) {
    if (arr[i].load + load > m) {
      return price
    } else if (arr[i].load + load === m) {
      return arr[i].price + price
    } else {
      arr.splice(0, 1)
      maxValue(arr, m, arr[i].load + load, arr[i].price + price)
    }
  }
}

function maxPrice(m, load, price) {
  var singlePrices = []
  for (var i = 0; i < load.length - 1; i++) {
    singlePrices.push({
      single: prices[i]/loads[i],
      load: loads[i],
      price: prices[i]
    })
  }
  var priceQuene = singlePrices.sort(function(a, b) {
    return b.single - a.single
  })      
  return maxValue(priceQuene, maxLoad, 0, 0)
}
console.log(maxPrice(maxLoad, loads, prices))
```
